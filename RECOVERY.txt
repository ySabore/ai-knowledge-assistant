Run API (macOS / Linux)
  cd backend
  python3 -m venv ../.venv-macos
  source ../.venv-macos/bin/activate   # Windows: ..\\.venv-macos\\Scripts\\activate
  pip install -r ../requirements-backend.txt
  export REGISTRY_DATABASE_PATH="$(cd .. && pwd)/app.db"   # optional if cwd is project root
  export DEV_BEARER_USER_ID=user-alpha                    # user resolved when using bearer token
  uvicorn ai_knowledge_assistant.main:app --reload --host 127.0.0.1 --port 8000

Or pin the user without a token (local only):
  export DEV_USER_ID=user-alpha

Run web (needs Node 18+)
  cd apps/web
  cp .env.example .env
  npm install
  npm run dev
  Open http://127.0.0.1:5173
